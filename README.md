```javascript
// A simple function to add two numbers
function add(a, b) {
  return a + b;
}

// Using the function
const num1 = 5;
const num2 = 10;
const sum = add(num1, num2);

console.log(`The sum of ${num1} and ${num2} is: ${sum}`); // Output: The sum of 5 and 10 is: 15

for (let i = 0; i < 5; i++) {
  console.log(`Iteration ${i}`);
```
